86static void board_lcd_write_words(uint16_t address, uint16_t *data, size_t size)
87{
88 // write register address
89 LCD_CLEAR_RS;
90 LCD_CLEAR_WR;
91 LCD_DATABUS_PORT->ODR = address;
92 LCD_SET_WR;
93 LCD_SET_RS;
94 // write register data
95 uint16_t *data_end = data + size;
96 while (data != data_end) {
97 LCD_DATABUS_PORT->ODR = *data;
98 LCD_CLEAR_WR;
99 LCD_SET_WR;

101 }

267void lcd_fill_region(int16_t x1, int16_t y1, int16_t x2, int16_t y2, uint16_t *colors_data)
268{
269 // set window
270 if (x1 > x2) {
271 int16_t x_tmp = x1;
272 x1 = x2;
273 x2 = x_tmp;
274 }
275 if (y1 > y2) {
276 int16_t y_tmp = y1;
277 y1 = y2;
278 y2 = y_tmp;
279 }
280 LCD_CAPTURE_BUS;
281 // set gram coordinates
282 bus_lcd_write_reg(LCD_REGISTER_GRAM_X_ADDR, x1);
283 bus_lcd_write_reg(LCD_REGISTER_GRAM_Y_ADDR, y1);
284 bus_lcd_write_reg(LCD_REGISTER_GRAM_X_ADDR_START, x1);
285 bus_lcd_write_reg(LCD_REGISTER_GRAM_X_ADDR_END, x2 - 1);
286 bus_lcd_write_reg(LCD_REGISTER_GRAM_Y_ADDR_START, y1);
287 bus_lcd_write_reg(LCD_REGISTER_GRAM_Y_ADDR_END, y2 - 1);
288
289 // write pixels
290 size_t total_pixes = (x2 - x1) * (y2 - y1);
291 board_lcd_write_words(LCD_REGISTER_GRAM_WRITE_DATA, colors_data, total_pixes);
292 LCD_RELEASE_BUS;
293}

295void lcd_fill_color(int16_t x1, int16_t y1, int16_t x2, int16_t y2, uint16_t color)
296{
297 // set window
298 if (x1 > x2) {
299 int16_t x_tmp = x1;
300 x1 = x2;
301 x2 = x_tmp;
302 }
303 if (y1 > y2) {
304 int16_t y_tmp = y1;
305 y1 = y2;
306 y2 = y_tmp;
307 }
308 // set gram coordinates
309 LCD_CAPTURE_BUS;
310 bus_lcd_write_reg(LCD_REGISTER_GRAM_X_ADDR, x1);
311 bus_lcd_write_reg(LCD_REGISTER_GRAM_Y_ADDR, y1);
312 bus_lcd_write_reg(LCD_REGISTER_GRAM_X_ADDR_START, x1);
313 bus_lcd_write_reg(LCD_REGISTER_GRAM_X_ADDR_END, x2 - 1);
314 bus_lcd_write_reg(LCD_REGISTER_GRAM_Y_ADDR_START, y1);
315 bus_lcd_write_reg(LCD_REGISTER_GRAM_Y_ADDR_END, y2 - 1);
316
317 // prepare buffer
318 size_t total_pixes = (x2 - x1) * (y2 - y1);
319 const size_t max_buff_size = 128;
320 size_t buff_size = total_pixes < max_buff_size ? total_pixes : max_buff_size;
321 uint16_t buff[buff_size];
322 for (size_t i = 0; i < buff_size; i++) {
323 buff[i] = color;
324 }
325
326 size_t i = total_pixes;
327 while (i > 0) {
328 size_t pixels_to_update = i >= buff_size ? buff_size : i;
329 // write pixels
330 board_lcd_write_words(LCD_REGISTER_GRAM_WRITE_DATA, buff, pixels_to_update);
331 i -= pixels_to_update;
332 }
333 LCD_RELEASE_BUS;
334}
