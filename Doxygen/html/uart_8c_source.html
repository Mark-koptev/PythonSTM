98void Init_UART2(void)
99{
100 /* (1) Enable GPIO clock and configures the USART2 pins ********************/
101 /* Enable the peripheral clock of GPIOC */
102 LL_AHB1_GRP1_EnableClock(LL_AHB1_GRP1_PERIPH_GPIOA);
103 /* Configure TX Pin connected to pin PA2 */
104 LL_GPIO_SetPinMode(GPIOA, LL_GPIO_PIN_2, LL_GPIO_MODE_ALTERNATE);
105 LL_GPIO_SetAFPin_0_7(GPIOA, LL_GPIO_PIN_2, LL_GPIO_AF_7);
106 LL_GPIO_SetPinSpeed(GPIOA, LL_GPIO_PIN_2, LL_GPIO_SPEED_FREQ_HIGH);
107 LL_GPIO_SetPinOutputType(GPIOA, LL_GPIO_PIN_2, LL_GPIO_OUTPUT_PUSHPULL);
108 LL_GPIO_SetPinPull(GPIOA, LL_GPIO_PIN_2, LL_GPIO_PULL_DOWN);
109
110 /* Configure RX Pin connected to pin PA3 */
111 LL_GPIO_SetPinMode(GPIOA, LL_GPIO_PIN_3, LL_GPIO_MODE_ALTERNATE);
112 LL_GPIO_SetAFPin_0_7(GPIOA, LL_GPIO_PIN_3, LL_GPIO_AF_7);
113 LL_GPIO_SetPinSpeed(GPIOA, LL_GPIO_PIN_3, LL_GPIO_SPEED_FREQ_HIGH);
114 LL_GPIO_SetPinOutputType(GPIOA, LL_GPIO_PIN_3, LL_GPIO_OUTPUT_PUSHPULL);
115 LL_GPIO_SetPinPull(GPIOA, LL_GPIO_PIN_3, LL_GPIO_PULL_UP);
116
117 /* (2) NVIC Configuration for USART2i interrupts */
118 /* - Set priority for USART2_IRQn */
119 /* - Enable USART2_IRQn */
120 NVIC_SetPriority(USART2_IRQn, 0);
121 NVIC_EnableIRQ(USART2_IRQn);
122
123 /* (3) Enable the USART2 peripheral clock and clock source ****************/
124 LL_APB1_GRP1_EnableClock(LL_APB1_GRP1_PERIPH_USART2);
125
126 /* (4) Configure USART2 functional parameters ********************************/
127
128 /* Disable USART prior modifying configuration registers */
129 /* Note: Commented as corresponding to Reset value */
130 LL_USART_Disable(USART2);
131
132 /* TX/RX direction */
133 LL_USART_SetTransferDirection(USART2, LL_USART_DIRECTION_TX_RX);
134
135 /* 8 data bit, 1 start bit, 1 stop bit, no parity */
136 LL_USART_ConfigCharacter(USART2, LL_USART_DATAWIDTH_8B, LL_USART_PARITY_NONE, LL_USART_STOPBITS_1);
137
138 /* Configure transfer bit order : MSB first */
139 LL_USART_SetTransferBitOrder(USART2, LL_USART_BITORDER_LSBFIRST);
140
141 /* Clock configuration : Phase 2 edges, Polarity Low, Last Bit Clock output enabled */
142 LL_USART_ConfigClock(USART2, LL_USART_PHASE_2EDGE, LL_USART_POLARITY_LOW, LL_USART_LASTCLKPULSE_OUTPUT);
143
144 /* Oversampling by 16 : To be forced to 16 in USART mode */
145 LL_USART_SetOverSampling(USART2, LL_USART_OVERSAMPLING_16);
146
147
148 /* Set Baudrate to 9600 using APB1 frequency set to 32000000 Hz */
149 /* Frequency available for USART peripheral can also be calculated through LL RCC macro */
150 /* Ex :
151 Periphclk = LL_RCC_GetUSARTClockFreq(Instance); or LL_RCC_GetUARTClockFreq(Instance); depending on USART/UART instance
152
153 In this example, Peripheral Clock is expected to be equal to 32000000 Hz => equal to SystemCoreClock/2
154 */
155 LL_USART_SetBaudRate(USART2, SystemCoreClock/2, LL_USART_OVERSAMPLING_16, 9600);
156
157 /* Enable USART2 RX Interrupt */
158 LL_USART_EnableIT_RXNE(USART2);
159 /* (5) Enable USART2 **********************************************************/
160 LL_USART_Enable(USART2);
161
162 /* Polling USART initialisation */
163 while((!(LL_USART_IsActiveFlag_TEACK(USART2))) || (!(LL_USART_IsActiveFlag_REACK(USART2))))
164 {
165 }
166 /* Enable USART2 ERROR Interrupt */
167 LL_USART_EnableIT_ERROR(USART2);
168 transmitRxbuf();
